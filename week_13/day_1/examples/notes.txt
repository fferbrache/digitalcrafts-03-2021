created a new file in reducers called counterReducer.js
const initialState = 0
const counter = (state=initialState.action) => {
  switch (action.typ) {
    case "INCREMENT":
      return state + 1
    case "DECREMENT":
      return state - 1
    default:
      return state
  }
};

export default counter;
